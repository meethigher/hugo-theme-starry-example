# 7. Dynamic Content (not recommended directly in Markdown)

<span id="workYear">99</span> years <span id="workDay">99</span> days <span id="workHour">99</span> hours <span id="workMinute">99</span> minutes <span id="workSecond">99</span> seconds

<script>
let year = 0, day = 0, hour = 0, minute = 0, second = 0;

function workTimer(timestamp) {
    if (timestamp > 0) {
        day = Math.floor(timestamp / (60 * 60 * 24));
        hour = Math.floor(timestamp / (60 * 60)) - (day * 24);
        minute = Math.floor(timestamp / 60) - (day * 24 * 60) - (hour * 60);
        second = Math.floor(timestamp) - (day * 24 * 60 * 60) - (hour * 60 * 60) - (minute * 60);

        let number = 365;
        if (day >= number) {
            year = Math.floor(day / number);
            day -= year * number;
        }
    }
    
    const yearElements = document.querySelectorAll(".y");
    yearElements.forEach(el => el.style.display = year === 0 ? "none" : "");
    
    const pad = n => n <= 9 ? "0"+n : n;
    const setText = (id, value) => { const el = document.getElementById(id); if(el) el.textContent=value; };
    
    setText("workYear", pad(year));
    setText("workDay", pad(day));
    setText("workHour", pad(hour));
    setText("workMinute", pad(minute));
    setText("workSecond", pad(second));
}

let beginTime = "2021-03-29 08:00:00";
setInterval(() => {
    let res = Math.floor(Date.now()/1000 - Date.parse(new Date(beginTime))/1000);
    workTimer(res);
}, 1000);
</script>

> Recommendation:  
> - Put dynamic content in **shortcode / JS file**  
> - Avoid placing directly in Markdown body
